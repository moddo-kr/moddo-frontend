import { Meta, ColorPalette, ColorItem } from '@storybook/blocks';
import theme from '../styles/theme';

<Meta title="Design Tokens/Colors" />

# Colors

## Primitives

<ColorPalette>
  {Object.entries(theme.color.primitive).map(([colorName, shades]) => {
    const isNested = Object.values(shades).some(value => typeof value === 'object');

    return isNested ? (
      Object.entries(shades).map(([subKey, values]) => (
        <ColorItem
          key={`${colorName}-${subKey}`}
          title={`${colorName}.${subKey}`}
          subtitle={`primitive.${colorName}.${subKey}`}
          colors={values}
        />
      ))
    ) : (
      <ColorItem key={colorName} title={`${colorName}`} subtitle={`primitive.${colorName}`} colors={shades} />
    );

})}

</ColorPalette>
